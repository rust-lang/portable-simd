5/// Choose elements from two vectors using a mask.
6///
7/// For each element in the mask, choose the corresponding element from `true_values` if
8/// that element mask is true, and `false_values` if that element mask is false.
9///
10/// If the mask is `u64`, it's treated as a bitmask with the least significant bit
11/// corresponding to the first element.
12///
13/// # Examples
14///
15/// ## Selecting values from `Simd`
16/// ```
17/// # #![feature(portable_simd)]
18/// # #[cfg(feature = "as_crate")] use core_simd::simd;
19/// # #[cfg(not(feature = "as_crate"))] use core::simd;
20/// # use simd::{Simd, Mask, Select};
21/// let a = Simd::from_array([0, 1, 2, 3]);
22/// let b = Simd::from_array([4, 5, 6, 7]);
23/// let mask = Mask::<i32, 4>::from_array([true, false, false, true]);
24/// let c = mask.select(a, b);
25/// assert_eq!(c.to_array(), [0, 5, 6, 3]);
26/// ```
27///
28/// ## Selecting values from `Mask`
29/// ```
30/// # #![feature(portable_simd)]
31/// # #[cfg(feature = "as_crate")] use core_simd::simd;
32/// # #[cfg(not(feature = "as_crate"))] use core::simd;
33/// # use simd::{Mask, Select};
34/// let a = Mask::<i32, 4>::from_array([true, true, false, false]);
35/// let b = Mask::<i32, 4>::from_array([false, false, true, true]);
36/// let mask = Mask::<i32, 4>::from_array([true, false, false, true]);
37/// let c = mask.select(a, b);
38/// assert_eq!(c.to_array(), [true, false, true, false]);
39/// ```
40///
41/// ## Selecting with a bitmask
42/// ```
43/// # #![feature(portable_simd)]
44/// # #[cfg(feature = "as_crate")] use core_simd::simd;
45/// # #[cfg(not(feature = "as_crate"))] use core::simd;
46/// # use simd::{Mask, Select};
47/// let a = Mask::<i32, 4>::from_array([true, true, false, false]);
48/// let b = Mask::<i32, 4>::from_array([false, false, true, true]);
49/// let mask = 0b1001;
50/// let c = mask.select(a, b);
51/// assert_eq!(c.to_array(), [true, false, true, false]);
52/// ```
53pub trait Select<T> {
54    /// Choose elements
55    fn select(self, true_values: T, false_values: T) -> T;
56}
57
58impl<T, U, const N: usize> Select<Simd<T, N>> for Mask<U, N>
59where
60    T: SimdElement,
61    U: MaskElement,
62    LaneCount<N>: SupportedLaneCount,
63{
64    #[inline]
65    fn select(self, true_values: Simd<T, N>, false_values: Simd<T, N>) -> Simd<T, N> {
66        // Safety:
67        // simd_as between masks is always safe (they're vectors of ints).
68        // simd_select uses a mask that matches the width and number of elements
69        unsafe {
70            let mask: Simd<T::Mask, N> = core::intrinsics::simd::simd_as(self.to_simd());
71            core::intrinsics::simd::simd_select(mask, true_values, false_values)
72        }
73    }
74}

76impl<T, const N: usize> Select<Simd<T, N>> for u64
77where
78    T: SimdElement,
79    LaneCount<N>: SupportedLaneCount,
80{
81    #[inline]
82    fn select(self, true_values: Simd<T, N>, false_values: Simd<T, N>) -> Simd<T, N> {
83        const {
84            assert!(N <= 64, "number of elements can't be greater than 64");
85        }
86
87        #[inline]
88        unsafe fn select_impl<T, U: FixEndianness, const M: usize, const N: usize>(
89            bitmask: U,
90            true_values: Simd<T, N>,
91            false_values: Simd<T, N>,
92        ) -> Simd<T, N>
93        where
94            T: SimdElement,
95            LaneCount<M>: SupportedLaneCount,
96            LaneCount<N>: SupportedLaneCount,
97        {
98            let default = true_values[0];
99            let true_values = true_values.resize::<M>(default);
100            let false_values = false_values.resize::<M>(default);
101
102            // LLVM assumes bit order should match endianness
103            let bitmask = bitmask.fix_endianness();
104
105            // Safety: the caller guarantees that the size of U matches M
106            let selected = unsafe {
107                core::intrinsics::simd::simd_select_bitmask(bitmask, true_values, false_values)
108            };
109
110            selected.resize::<N>(default)
111        }
112
113        // TODO modify simd_bitmask_select to truncate input, making this unnecessary
114        if N <= 8 {
115            let bitmask = self as u8;
116            // Safety: bitmask matches length
117            unsafe { select_impl::<T, u8, 8, N>(bitmask, true_values, false_values) }
118        } else if N <= 16 {
119            let bitmask = self as u16;
120            // Safety: bitmask matches length
121            unsafe { select_impl::<T, u16, 16, N>(bitmask, true_values, false_values) }
122        } else if N <= 32 {
123            let bitmask = self as u32;
124            // Safety: bitmask matches length
125            unsafe { select_impl::<T, u32, 32, N>(bitmask, true_values, false_values) }
126        } else {
127            let bitmask = self;
128            // Safety: bitmask matches length
129            unsafe { select_impl::<T, u64, 64, N>(bitmask, true_values, false_values) }
130        }
131    }
132}

134impl<T, U, const N: usize> Select<Mask<T, N>> for Mask<U, N>
135where
136    T: MaskElement,
137    U: MaskElement,
138    LaneCount<N>: SupportedLaneCount,
139{
140    #[inline]
141    fn select(self, true_values: Mask<T, N>, false_values: Mask<T, N>) -> Mask<T, N> {
142        let selected: Simd<T, N> =
143            Select::select(self, true_values.to_simd(), false_values.to_simd());
144
145        // Safety: all values come from masks
146        unsafe { Mask::from_simd_unchecked(selected) }
147    }
148}
